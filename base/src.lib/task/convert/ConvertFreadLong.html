ConvertFreadLong

Syntax
int ConvertFreadLong(FILE *fp,int64 *val);
Header
base/rconvert.h
Library
rcnv
Description

The ConvertFreadLong function reads a variable of type int64 (64-bit integer) from an open I/O stream. The bytes are assumed to be in little-endian ordered.

The bytes are read from the stream pointed to by the argument fp and the 64-bit integer number is stored at the location pointed to by the argument val.

Returns
Returns zero on success. On error, a (-1) is returned.
Errors
On error, a (-1) is returned.

#include <stdio.h>
#include <stdlib.h>
#include "rtypes.h"
#include "rconvert.h"

#define FNAME "test.dat"

int main(int argc,char *argv[]) {
  int64 val;
  FILE *fp;  
  fp=fopen(FNAME,"r");
   
  if (fp==NULL) {
    fprintf(stderr,"File not found.'n");
    exit(-1);
  }
  
  while (ConvertFreadLong(fp,&val)==0) {
    fprintf(stdout,"%lld'n",val);
  }   
  fclose(fp);
  
  return 0;
}
