ConvertFromShort

Syntax
void ConvertFromShort(int16 val,unsigned char *cnv);
Header
base/rconvert.h
Library
rcnv
Description

The ConvertFromShort function converts from a variable of type int16 (16-bit integer) to a sequence of little-endian ordered bytes.

The number to convert is given by the argument val and the sequence of bytes are stored at the location pointed to by cnv.

Returns
None
Errors
None

#include <stdio.h>
#include <stdlib.h>
#include "rtypes.h"
#include "rconvert.h"

int main(int argc,char *argv[]) {
  int i;
  int16 val;
  unsigned char buf[4];
  
  val=64;
  ConvertFromShort(val,buf);
  
  fprintf(stdout,"val=%d'n",val);
  fprintf(stdout,"buf=");
  for (i=0;i<2;i++) fprintf(stdout,"%.2x",buf[i]);
  fprintf(stdout,"'n");
  return 0;
}
