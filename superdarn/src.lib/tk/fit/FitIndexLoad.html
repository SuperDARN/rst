FitIndexLoad

Syntax
struct FitIndex *FitIndexLoad(int fid);
Header
superdarn/fitindex.h
Library
fit
Description

The FitIndexLoad function loads an index to a fitacf data file.

The index is loaded from the file with the descriptor given by the argument fid.

Returns
Returns a pointer to a structure containing the index on success. On error, a NULL pointer is returned.
Errors
On error, a NULL pointer is returned.

#include <stdio.h>
#include <stdlib.h>

#include "rtypes.h"
#include "rtime.h"
#include "dmap.h"
#include "limit.h"
#include "rprm.h"
#include "fitdata.h"
#include "fitread.h"
#include "fitindex.h"
#include "fitseek.h"

int main(int argc,char *argv[]) {
 
  FILE *fp;
  int fd=0;

  struct RadarParm prm;
  struct FitData fit;
  struct FitIndex *inx=NULL;



  int yr=2003,mo=5,dy=28,hr=12,mt=0;
  double sc=0,atme;

  int s;

  if (argc>2) {
     
    fp=fopen(argv[2],"r");

    if (fp==NULL) {
      fprintf(stderr,"Index not found.'n");
      exit(-1);
    }
    fd=fileno(fp);
    inx=FitIndexLoad(fd);
  
    fclose(fp);
    if (inx==NULL) {
        fprintf(stderr,"Error loading index.'n");
        exit(-1);
    }
  }

  fp=fopen(argv[1],"r");

  if (fp==NULL) {
    fprintf(stderr,"File not found.'n");
    exit(-1);
  }
  fd=fileno(fp);


  s=FitSeek(fd,yr,mo,dy,hr,mt,(int) sc,&atme,inx);
  if (s==-1) {
    fprintf(stderr,"file does not contain that interval.'n");
    exit(-1);
  }

  fprintf(stdout,"Requested:%.4d-%.2d-%.2d %.2d:%.2d:%.2d'n",
            yr,mo,dy,hr,mt,(int) sc);

  TimeEpochToYMDHMS(atme,&yr,&mo,&dy,&hr,&mt,&sc);
  fprintf(stdout,"Found:%.4d-%.2d-%.2d %.2d:%.2d:%.2d'n",
            yr,mo,dy,hr,mt,(int) sc);
 

  while(FitRead(fd,&prm,&fit) !=-1) {
    fprintf(stdout,"%.4d-%.2d-%.2d %.2d:%.2d:%.2d'n",
             prm.time.yr,prm.time.mo,prm.time.dy,
             prm.time.hr,prm.time.mt,prm.time.sc);
  }
 
  fclose(fp);


  return 0;
}
