GridFwrite

Syntax
int GridFwrite(FILE *fp,struct GridData *grd);
Header
superdarn/gridwrite.h
Library
grid
Description

The GridFwrite function writes a gridded data to an open stream.

The data is written to the open stream pointed to by the argument fp.

The gridded data is taken from the structure pointed to by the argument grd.

Returns
Returns the number of bytes written on success. On error, (-1) is returned.
Errors
On error, (-1) is returned.

#include <stdio.h>
#include <stdlib.h>

#include "rtypes.h"
#include "rtime.h"
#include "dmap.h"
#include "limit.h"
#include "griddata.h"
#include "gridread.h"
#include "gridwrite.h"

  struct GridData grd;


int main(int argc,char *argv[]) {
 
  FILE *fp;

  int yr,mo,dy,hr,mt;
  double sc;

  fp=fopen(argv[1],"r");

  if (fp==NULL) {
    fprintf(stderr,"File not found.'n");
    exit(-1);
  }

  while(GridFread(fp,&grd) !=-1) {

    TimeEpochToYMDHMS(grd.st_time,&yr,&mo,&dy,&hr,&mt,&sc);
    fprintf(stderr,"%.4d-%.2d-%.2d %.2d:%.2d:%.2d pnts=%d'n",
            yr,mo,dy,hr,mt,(int) sc,grd.vcnum);

    
    GridFwrite(stdout,&grd);
    
  }
 
  fclose(fp);


  return 0;
}
